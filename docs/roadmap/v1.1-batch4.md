## Unit 1 — #342: API root discovery endpoint

**Branch:** `fix/342-api-discovery`

### Background

`JsonApiRouteProvider::registerRoutes()` loops over entity types and registers five routes each,
but never registers a root `/api` route. Adding one requires:
1. A controller method that queries the `EntityTypeManager` and returns a JSON:API-style meta
   document with links to each entity type's collection endpoint.
2. A route registered before the entity loop in `registerRoutes()`.

The discovery document format (following JSON:API spec `meta` + `links`):
```json
{
  "meta": {
    "api": "waaseyaa",
    "version": "1.0"
  },
  "links": {
    "self": "/api",
    "node": { "href": "/api/node", "meta": { "type": "node" } },
    "user": { "href": "/api/user", "meta": { "type": "user" } }
  }
}
```

```php
<?php

declare(strict_types=1);

namespace Waaseyaa\Api;

use Waaseyaa\Entity\EntityTypeManagerInterface;

/**
 * Handles GET /api — returns a JSON:API-style discovery document.
 *
 * Lists all registered entity types with their collection endpoint URLs.
 */
final class ApiDiscoveryController
{
    public function __construct(
        private readonly EntityTypeManagerInterface $entityTypeManager,
        private readonly string $basePath = '/api',
    ) {}

    /**
     * Returns a discovery document describing all available entity type endpoints.
     *
     * @return array{meta: array<string, string>, links: array<string, mixed>}
     */
    public function discover(): array
    {
        $links = ['self' => $this->basePath];

        foreach ($this->entityTypeManager->getDefinitions() as $id => $definition) {
            $links[$id] = [
                'href' => $this->basePath . '/' . $id,
                'meta' => ['type' => $id],
            ];
        }

        return [
            'meta' => [
                'api' => 'waaseyaa',
                'version' => '1.0',
            ],
            'links' => $links,
        ];
    }
}
```

### Background

`PipelineExecutor` is a pure PHP class that takes an array of `PipelineStepInterface` plugins
and executes them in weight order. `SchemaRegistry` wraps `EntityJsonSchemaGenerator` +
`McpToolGenerator`. Both are fully testable with in-memory setup — no external API calls needed.

Key classes:
- `Waaseyaa\AI\Pipeline\PipelineExecutor` — takes `array<string, PipelineStepInterface>`
- `Waaseyaa\AI\Pipeline\Pipeline` — entity with `getSteps(): array<PipelineStepConfig>`
- `Waaseyaa\AI\Pipeline\PipelineStepConfig` — `id`, `pluginId`, `weight`, `configuration`
- `Waaseyaa\AI\Pipeline\PipelineContext` — `pipelineId`, `startedAt`, `get/set`
- `Waaseyaa\AI\Pipeline\StepResult` — `::success(output)`, `::failure(message)`
- `Waaseyaa\AI\Schema\SchemaRegistry` — `getSchema(id)`, `getAllSchemas()`, `getTools()`
- `Waaseyaa\AI\Schema\EntityJsonSchemaGenerator` — `generate(id)`, `generateAll()`

For `SchemaRegistry` tests, wire `EntityJsonSchemaGenerator` with a real `EntityTypeManager`
(from `packages/entity/src/EntityTypeManager.php`) using `Symfony\Component\EventDispatcher\EventDispatcher`.

For `PipelineExecutor` tests, use inline anonymous classes implementing `PipelineStepInterface`.

```php
<?php

declare(strict_types=1);

namespace Waaseyaa\Tests\Integration\Phase24;

use PHPUnit\Framework\Attributes\CoversNothing;
use PHPUnit\Framework\Attributes\Test;
use PHPUnit\Framework\TestCase;
use Waaseyaa\AI\Pipeline\Pipeline;
use Waaseyaa\AI\Pipeline\PipelineContext;
use Waaseyaa\AI\Pipeline\PipelineExecutor;
use Waaseyaa\AI\Pipeline\PipelineStepConfig;
use Waaseyaa\AI\Pipeline\PipelineStepInterface;
use Waaseyaa\AI\Pipeline\StepResult;
use Waaseyaa\AI\Schema\EntityJsonSchemaGenerator;
use Waaseyaa\AI\Schema\Mcp\McpToolGenerator;
use Waaseyaa\AI\Schema\SchemaRegistry;
use Waaseyaa\Entity\EntityType;
use Waaseyaa\Entity\EntityTypeManager;
use Symfony\Component\EventDispatcher\EventDispatcher;

/**
 * Integration tests for AI pipeline packages.
 *
 * Tests PipelineExecutor with real step plugins and SchemaRegistry with
 * real EntityTypeManager. No external services required.
 */
#[CoversNothing]
final class AIPipelineIntegrationTest extends TestCase
{
    // ── PipelineExecutor tests ──────────────────────────────────────────────

    #[Test]
    public function executor_runs_steps_in_order_and_chains_output(): void
    {
        $log = [];

        $stepA = new class ($log) implements PipelineStepInterface {
            public function __construct(private array &$log) {}
            public function process(array $input, PipelineContext $ctx): StepResult
            {
                $this->log[] = 'A';
                return StepResult::success([...$input, 'a' => true]);
            }
        };

        $stepB = new class ($log) implements PipelineStepInterface {
            public function __construct(private array &$log) {}
            public function process(array $input, PipelineContext $ctx): StepResult
            {
                $this->log[] = 'B';
                return StepResult::success([...$input, 'b' => true]);
            }
        };

        $pipeline = new Pipeline(['steps' => [
            new PipelineStepConfig(id: 'step-a', pluginId: 'plugin_a', weight: 10),
            new PipelineStepConfig(id: 'step-b', pluginId: 'plugin_b', weight: 20),
        ]]);

        $executor = new PipelineExecutor(stepPlugins: [
            'plugin_a' => $stepA,
            'plugin_b' => $stepB,
        ]);

        $result = $executor->execute($pipeline, ['seed' => 1]);

        $this->assertTrue($result->success);
        $this->assertSame(['A', 'B'], $log);
        $this->assertSame(['seed' => 1, 'a' => true, 'b' => true], $result->finalOutput);
        $this->assertSame('Pipeline completed successfully.', $result->message);
    }

    #[Test]
    public function executor_stops_and_reports_failure_when_step_fails(): void
    {
        $failingStep = new class implements PipelineStepInterface {
            public function process(array $input, PipelineContext $ctx): StepResult
            {
                return StepResult::failure('Upstream service unavailable.');
            }
        };

        $shouldNotRun = new class implements PipelineStepInterface {
            public bool $ran = false;
            public function process(array $input, PipelineContext $ctx): StepResult
            {
                $this->ran = true;
                return StepResult::success($input);
            }
        };

        $pipeline = new Pipeline(['steps' => [
            new PipelineStepConfig(id: 'fail', pluginId: 'failing', weight: 10),
            new PipelineStepConfig(id: 'after', pluginId: 'after_fail', weight: 20),
        ]]);

        $executor = new PipelineExecutor(stepPlugins: [
            'failing'    => $failingStep,
            'after_fail' => $shouldNotRun,
        ]);

        $result = $executor->execute($pipeline);

        $this->assertFalse($result->success);
        $this->assertStringContainsString('Upstream service unavailable.', $result->message);
        $this->assertFalse($shouldNotRun->ran);
    }

    #[Test]
    public function executor_fails_immediately_when_plugin_is_missing(): void
    {
        $pipeline = new Pipeline(['steps' => [
            new PipelineStepConfig(id: 'step-1', pluginId: 'nonexistent_plugin', weight: 10),
        ]]);

        $executor = new PipelineExecutor(stepPlugins: []);

        $result = $executor->execute($pipeline, ['x' => 1]);

        $this->assertFalse($result->success);
        $this->assertStringContainsString('nonexistent_plugin', $result->message);
    }

    #[Test]
    public function executor_returns_success_for_empty_pipeline(): void
    {
        $pipeline = new Pipeline(['steps' => []]);
        $executor = new PipelineExecutor(stepPlugins: []);

        $result = $executor->execute($pipeline, ['data' => 'x']);

        $this->assertTrue($result->success);
        $this->assertSame('Pipeline has no steps.', $result->message);
        $this->assertSame(['data' => 'x'], $result->finalOutput);
    }

    // ── SchemaRegistry tests ────────────────────────────────────────────────

    private function makeRegistry(EntityTypeManager $manager): SchemaRegistry
    {
        $generator    = new EntityJsonSchemaGenerator($manager);
        $toolGenerator = new McpToolGenerator($manager);
        return new SchemaRegistry($generator, $toolGenerator);
    }

    #[Test]
    public function schema_registry_returns_valid_json_schema_for_entity_type(): void
    {
        $manager = new EntityTypeManager(new EventDispatcher());
        $manager->addDefinition(new EntityType(
            id: 'article',
            label: 'Article',
            entityClass: \stdClass::class,
            entityKeys: ['id' => 'nid', 'uuid' => 'uuid', 'label' => 'title'],
        ));

        $registry = $this->makeRegistry($manager);
        $schema   = $registry->getSchema('article');

        $this->assertSame('https://json-schema.org/draft/2020-12/schema', $schema['$schema']);
        $this->assertSame('Article', $schema['title']);
        $this->assertArrayHasKey('nid', $schema['properties']);
        $this->assertArrayHasKey('uuid', $schema['properties']);
        $this->assertArrayHasKey('title', $schema['properties']);
        $this->assertContains('nid', $schema['required']);
    }

    #[Test]
    public function schema_registry_returns_all_schemas(): void
    {
        $manager = new EntityTypeManager(new EventDispatcher());
        $manager->addDefinition(new EntityType(
            id: 'page',
            label: 'Page',
            entityClass: \stdClass::class,
            entityKeys: ['id' => 'id'],
        ));
        $manager->addDefinition(new EntityType(
            id: 'event',
            label: 'Event',
            entityClass: \stdClass::class,
            entityKeys: ['id' => 'id'],
        ));

        $registry = $this->makeRegistry($manager);
        $all      = $registry->getAllSchemas();

        $this->assertArrayHasKey('page', $all);
        $this->assertArrayHasKey('event', $all);
        $this->assertSame('Page', $all['page']['title']);
    }
}
```
